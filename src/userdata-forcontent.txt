#!/bin/bash
cd /home/ubuntu/

sudo apt update -y

sudo apt install git

sudo apt install awscli
aws configure set default.region us-east-1
aws configure set default.output json
access_key_id=$(aws ssm get-parameter --name "/myapp/aws_access_key_id" --query "Parameter.Value" --with-decryption --output text)
secret_access_key=$(aws ssm get-parameter --name "/myapp/aws_secret_access_key" --query "Parameter.Value" --with-decryption --output text)
echo "[default]" >> /home/ec2-user/.aws/credentials
echo "aws_access_key_id = $access_key_id" >> /home/ec2-user/.aws/credentials
echo "aws_secret_access_key = $secret_access_key" >> /home/ec2-user/.aws/credentials

mkdir audio_files
cd audio_files
mkdir wavs
mkdir split_wavs
cd ..
mkdir video_scripts
cd video_scripts
git clone https://github.com/djtoler/Python-Content-Creation-Scripts.git
cd ..

sudo apt  install ffmpeg -y

curl -sL https://deb.nodesource.com/setup_16.x | sudo bash -

$ cat /etc/apt/sources.list.d/nodesource.list
deb https://deb.nodesource.com/node_16.x focal main
deb-src https://deb.nodesource.com/node_16.x focal main


sudo apt -y install nodejs
sudo apt -y install gcc g++ make

sudo npm install pm2@latest -g
sudo pm2 start index.js
sudo pm2 startup
sudo pm2 save

export HOME=/home/ubuntu
wget https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh
bash Miniconda3-latest-Linux-x86_64.sh -b
echo 'export PATH="/home/ubuntu/miniconda3/bin:$PATH"' >> ~/.bashrc
source ~/.bashrc
source /home/ubuntu/miniconda3/etc/profile.d/conda.sh
conda update -n base -c conda-forge conda -y
source ~/.bashrc
conda init bash




conda create --name newenv python=3.9 -y
conda activate newenv 
conda install pip
sudo apt install awscli -y
conda install numpy scipy matplotlib pandas jupyter
pip install scikit-learn
pip install hmmlearn
pip install imbalanced-learn
pip install plotly

python3 -m pip install -U demucs
pip install eyed3
pip install pyAudioAnalysis
pip install pydub 
pip install moviepy
pip install youtube-download-cli
python -m pip install git+https://github.com/pytube/pytube
